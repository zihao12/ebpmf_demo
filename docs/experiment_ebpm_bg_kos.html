K = ncol(L)
I construct a poisson means sub-problem this way \[\begin{align} & Y_{jk} = \sum_i Z_{ijk}\\ & Z_{ijk} = X_{ij} \zeta_{ijk}\\ & \zeta_{ijk} = \frac{l_{ik} f_{jk}}{\sum_k l_{ik} f_{jk}}\\ \end{align}\]

Then fit ebpm_background model on \(Y_{jk}\)

## get Y
get_Y <- function(X,L, F ,K){
  n = nrow(X)
  p = ncol(X)
  ## zeta = array(dim = c(n, p, K))
  Z_sum = array(, dim = c(p, K))
  Lam = L %*% t(F)
  for(k in 1:K){
    #Z_sum[,k] = colSums(X * ((L[,k] %o% F[,k])/Lam))
    tmp = X * (L[,k] %o% F[,k])
    mask <- (tmp != 0)
    tmp[mask] = tmp[mask]/Lam[mask]
    Z_sum[,k] = colSums(tmp)
  }
  return(Z_sum)
}

Y = get_Y(X = X, L = L, F = F, K = K)

## fit model
maxiter = 100
a_L = c(seq(0.01, 0.10, 0.01), seq(0.2, 0.9, 0.1), seq(1,15,2), 20, 50, 75, 100, 200, 1e3)
grids= list(al = a_L, bl = a_L)

Lam = t(t(F) * s_k)
runtime <- system.time(
  fit <- ebpm_background(Y = Y, s = replicate(K, 1),
                grids = grids, maxiter = maxiter)
)
